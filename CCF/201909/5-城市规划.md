

#### Dijkstra的0分代码
```
#include <bits/stdc++.h>
using namespace std;

const int MAXV = 2010;
const int inf = 0x3f3f3f3f;
int G[MAXV][MAXV];
int d[MAXV];
bool vis[MAXV] = { false };
int n,m,k;
int imp[2010];

void Dijkstra(int s) {
	fill(d, d + MAXV, inf);
	d[s] = 0;
	for (int i = 0; i < n; i++) {
		int u = -1, MIN = inf;
		for (int j = 1; j <= n; j++) {
			if (vis[j] == false && d[j] < MIN) {
				u = j;
				MIN = d[j];
			}
		}
		if (u == -1)
			return;
		vis[u] = true;
		for (int v = 1; v <= n; v++) {
			if (vis[v] == false && G[u][v] != inf) {
				if (d[u] + G[u][v] < d[v]) {
					d[v] = d[u] + G[u][v];
				}
			}
		}
	}
}


int main(){
    scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<m;i++)
        scanf("%d",&imp[i]);
    fill(G[0],G[0]+MAXV*MAXV,inf);
    for(int i=0;i<n-1;i++){
        int u,v,c;
        scanf("%d%d%d",&u,&v,&c);
        G[u][v]=G[v][u]=c;
    }

    int minn=inf;
    if(k==1){
        printf("0");
        return 0;
    }
    for(int i=0;i<m;i++){
        Dijkstra(imp[i]);
        for(int j=0;j<m;j++){
            if(i==j)
                continue;
            if(d[imp[j]]==0)
                continue;
            minn=min(minn,d[imp[j]]);
        }
    }

    printf("%d\n",minn);
    return 0;
}
```
