```cpp
void split(string &p,vector<string> &v){
    for(int i=0,j;i<p.size();i=j){
        j=p.find('/',i);
        if(j==string::npos)
            v.push_back(p.substr(i));
        else{

    }
}
```

```cpp
#include <iostream>
#include <sstream>
#include <vector>

using namespace std;

int main(){
    int n;
    scanf("%d",&n);
    string path,s,str;
    ///path:存根目录
    ///s:
    ///str:存读取的字符串
    cin>>path;
    getchar();
    for(int i=0;i<n;i++){
        getline(cin,str);
        if(str=="") str=path;
        if(str[0]!='/') str=path+'/'+str;///说明这个路径是相对路径
        for(int j=0;j<str.size();j++)
            if(str[j]=='/') str[j]=' ';
        vector<string> sta;
        stringstream ss(str);

        while(ss>>s){
            if(s==".")  continue;           ///本级路径
            else if(s==".."&&!sta.empty())  sta.pop_back();///上一级路径
            else if(s!="..")    sta.push_back(s);         ///不是上一级路径也不是本级路径，则推入该路径
        }
        cout<<'/';
        for(int j=0;j<sta.size();j++){
            if(j)   cout<<'/';
            cout<<sta[j];
        }
        cout<<endl;
    }
    return 0;
}
```
