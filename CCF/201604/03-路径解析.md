#include <bits/stdc++.h>

using namespace std;

void split(string &p,vector<string> &v){
    for(int i=0,j;i<p.size();i=j){
        j=p.find('/',i);
        if(j==string::npos)
            v.push_back(p.substr(i));
        else{
            printf("未到最后,i:%d,j:%d\n",i,j);
            v.push_back(p.substr(i,j-i));
        }
        while(j<p.size()&&p[j]=='/')
            ++j;
    }
}

int main(){
    string path;
    getline(cin,path);
    vector<string> vec;
    split(path,vec);
    for(int i=0;i<vec.size();i++)
        cout<<"第"<<i<<"个:"<<vec[i]<<endl;

    return 0;
}
