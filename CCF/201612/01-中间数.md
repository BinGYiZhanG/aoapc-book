```
#include <iostream>

using namespace std;

int main()
{
    int a[1010];
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);
    int cnt=-1;
    for(int i=0;i<n;i++){
        int lf=0,rt=0;
        for(int j=0;j<n;j++){
            if(i==j)
                continue;
            if(a[i]>a[j])
                rt++;
            if(a[i]<a[j])
                lf++;
        }
        if(lf==rt)
            cnt=i;
    }
    printf("%d\n",cnt);
    return 0;
}
```
