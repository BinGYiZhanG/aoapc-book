* 钥匙盒一共有N个挂钩，从左到右排成一排，用来挂N个教室的钥匙。
* 有K位老师要上课，给出每位老师所需要的钥匙、开始上课的时间和上课的时长，假设下课时间就是还钥匙时间，请问最终钥匙盒里面钥匙的顺序是怎样的？


* 10分代码

```
#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

struct Tea{
    int num,st,tr;///开始时间，持续时间
}tea[1010];

int key[1010];
bool cmp_st(Tea t1,Tea t2){
    return t1.st<t2.st;
}

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    int maxtime=0;
    for(int i=1;i<=n;i++)
        key[i]=i;
    for(int i=0;i<k;i++){
        cin>>tea[i].num>>tea[i].st>>tea[i].tr;
        maxtime=max(maxtime,tea[i].st+tea[i].tr);
    }
    sort(tea,tea+k,cmp_st);

/*
    for(int i=0;i<k;i++){
        cout<<tea[i].num<<" "<<tea[i].st<<" "<<tea[i].tr<<endl;
    }
*/

    for(int i=1;i<=maxtime;i++){///时间,
        int u=-1;
        for(int j=0;j<k;j++)
            if(i==tea[j].st+tea[j].tr){
                u=j;
                break;
            }

        ///如果u=-1,没有老师在此刻还钥匙
        ///还钥匙操作
        for(int j=1;j<=n;j++)
        if(key[j]==0&&u!=-1){
            key[j]=tea[u].num;
            break;
        }

        u=-1;
        for(int j=0;j<k;j++)///钥匙
            if(i==tea[j].st){
                u=j;
                break;
            }
        ///如果u=-1,没有老师在此刻拿走钥匙
        if(u==-1)
            continue;
        for(int j=1;j<=n;j++)
        if(key[j]==tea[u].num){
            key[j]=0;
            break;
        }
        for(int i=1;i<=n;i++)
            printf("%d ",key[i]);
        cout<<endl;
    }

    for(int i=1;i<=n;i++)
        printf("%d ",key[i]);

    return 0;
}
```
