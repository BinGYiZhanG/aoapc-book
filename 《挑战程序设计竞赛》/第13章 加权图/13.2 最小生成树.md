#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
using namespace std;

const int inf=0x3f3f3f3f;
int G[110][110];
int d[110];
bool vis[110];
int n;

void Dijsktra(){
    d[0]=0;
    for(int i=0;i<n;i++){
        int u=-1,minn=inf;
        for(int j=0;j<n;j++){
            if(d[u]<minn&&vis[u]==false){
                u=j;
                minn=d[u];
            }
        }
        if(u==-1)
            return ;
        vis[u]=true;
        for(int v=0;v<n;v++){
            if(vis[v]==false&&G[u][v]!=-1&&d[u]+G[u][v]<d[v]){
                d[v]=d[u]+G[u][v];
            }
        }
    }
}


int main(){
    //fill(G[0],G[0]+110*110,inf);
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            scanf("%d",&G[i][j]);
        }
    }
    //memset(d,0,sizeof(d));
    memset(vis,0,sizeof(vis));
    fill(d,d+110,inf);
    Dijsktra();
    for(int i=0;i<n;i++){
        printf("%d:%d\n",i,d[i]);
    }
    printf("%d\n",d[n-1]);
    return 0;
}
