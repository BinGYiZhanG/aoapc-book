//#define LOCAL
#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

const int maxn=1010;

char mp[maxn][2];
bool vis[maxn];
int cnt=0;

bool DFS(int pos){
    vis[pos]=true;
    if(mp[pos][1]=='m') return true;
    for(int i=0;i<cnt;i++){
        if(vis[i]!=true&&mp[i][0]==mp[pos][1]){
            if(DFS(i))
                return true;
        }
    }
    return false;
}

int main(){
    char str[15];
    while(scanf("%s",str)==1&&str[0]!='0'){
        int len=strlen(str);
        mp[cnt][0]=str[0];
        mp[cnt][1]=str[len-1];
        cnt++;
        getchar();
    }
    bool res=false;
    fill(vis,vis+maxn,false);
    for(int i=0;i<cnt;i++){
        if(mp[i][0]=='b'&&vis[i]!=true){
            if(DFS(i)){
                res=true;
                break;
            }
        }
    }
    if(res)
        printf("Yes.\n");
    else
        printf("No.\n");
    return 0;
}
