```cpp
#include<cstdio>

using namespace std;
const int N=505;
int line[N][N];
int girl[N],used[N];
int k,m,n;
bool found(int x)
{
    for(int i=1; i<=m; i++)
    {
        if(line[i][x]&&!used[i])
        {
            used[i]=1;
            if(girl[i]==0||found(girl[i]))
            {
                girl[i]=x;
                return 1;
            }
        }
    }
    return 0;
}
int main()
{
    int x,y;
    scanf("%d %d",&m,&n);///外籍飞行员人数m,编号1~m
                        ///英国飞行员人数n,编号m+1~n
    memset(line,0,sizeof(line));
    memset(girl,0,sizeof(girl));
    while(scanf("%d%d",&x,&y)==2){
        if(x==-1&&y==-1)
            break;
        line[x][y]=1;
    }


    int sum=0;
    for(int i=m+1; i<=m+n; i++)
    {
        memset(used,0,sizeof(used));
        if(found(i)) sum++;
    }
    if(sum==0)
        printf("No Solution!\n");
    else{
        printf("%d\n",sum);
        for(int i=1;i<=m;i++){
            if(girl[i]!=0)
                printf("%d %d\n",i,girl[i]);
        }
    }
    return 0;
}
```

```cpp
#include <iostream>
#include <cstdio>
#include <queue>
#include <cstring>
#include <vector>

using namespace std;

const int maxn=100;
const int maxm=30000+5;
const int inf=0x3f3f3f3f;

struct Edge{
    int from,to,cap,flow;
    Edge(){}
    Edge(int from,int to,int cap,int flow){
        this->from=from;
        this->to=to;
        this->cap=cap;
        this->flow=flow;
    }
};

struct Dinic{
    int n,m,s,t;
    Edge edges[maxm];///边集
    int head[maxn];///头
    int nxt[maxm];///下一个结点
    bool vis[maxn];///
    int d[maxn];
    int cur[maxn];///不知何意

    void init(int n){
        this->n=n;
        memset(head,-1,sizeof(head));
        m=0;
    }

    void AddEdge(int u,int v,int c){
        edges[m]=Edge(u,v,c,0);
        nxt[m]=head[u];
        head[u]=m++;
        edges[m]=Edge(v,u,c,0);
        nxt[m]=head[v];
        head[v]=m++;
    }

    bool BFS(){
        memset(vis,0,sizeof(vis));
        queue<int> q;
        q.push(s);
        d[s]=0;
        vis[s]=1;
        while(!q.empty()){
            int x=q.front();q.pop();
            for(int i=head[x];i!=-1;i=nxt[i]){
                Edge& e=edges[i];
                if(!vis[e.to]&&e.cap>e.flow){
                    vis[e.to]=1;
                    d[e.to]=d[x]+1;
                    q.push(e.to);
                }
            }
        }
        return vis[t];
    }
    int DFS(int x,int a){
        if(x==t||a==0)  return a;
        int flow=0,f;
        for(int& i=cur[x];i!=-1;i=nxt[i]){
            Edge& e=edges[i];
            if(d[x]+1==d[e.to]&&(f=DFS(e.to,min(a,e.cap-e.flow)))>0){
            ///结点x的下一层是e.to,,   e.to结点被访问过了
                e.flow+=f;///边容量增加
                edges[i^1].flow-=f;///反向边容量减少
                flow+=f;
                a-=f;
                if(a==0)    break;
            }
        }
        return flow;
    }

    int MaxFlow(int s,int t){
        this->s=s;this->t=t;
        int flow=0;
        while(BFS()){
            for(int i=0;i<m;i++)
                cur[i]=head[i];///可能是每次都得更新head数组,
            flow+=DFS(s,inf);
        }
        return flow;
    }
}H;


int main()
{
    int n,m,u,v;
   // scanf("%d",&t);
  //  for(int ii=1;ii<=t;ii++){
    scanf("%d%d",&m,&n);
    H.init(n+3);

    while(scanf("%d%d",&u,&v)==2){
        if(u==-1&&v==-1)
            break;
        H.AddEdge(u,v,inf);
    }
    for(int i=1;i<=m;i++)///外籍飞行员编号
        H.AddEdge(i,n+1,1);///源点:n+2
    for(int i=m+1;i<=m+n;i++)///英国飞行员编号
        H.AddEdge(n+2,i,1);///汇点:n+1

    //}
    int ans=H.MaxFlow(n+2,n+1);
    if(!ans)
        printf("No Solution!\n");
    else{
        printf("%d\n",ans);
        for(int i=0;i<=H.m;i+=2)
            if(H.edges[i].to!=n+1&&
               H.edges[i].to!=n+2&&
               H.edges[i].from!=n+1&&
               H.edges[i].from!=n+2&&
               H.edges[i^1].flow
               )
               printf("%d %d\n",H.edges[i].from,H.edges[i].to);
    }

    return 0;
}
```
