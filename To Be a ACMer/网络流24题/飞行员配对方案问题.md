```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;
const int N=505;
int line[N][N];
int girl[N],used[N];
int k,m,n;
bool found(int x)
{
    for(int i=1; i<=m; i++)
    {
        if(line[i][x]&&!used[i])
        {
            used[i]=1;
            if(girl[i]==0||found(girl[i]))
            {
                girl[i]=x;
                return 1;
            }
        }
    }
    return 0;
}
int main()
{
    int x,y;
    scanf("%d %d",&m,&n);///外籍飞行员人数m,编号1~m
                        ///英国飞行员人数n,编号m+1~n
    memset(line,0,sizeof(line));
    memset(girl,0,sizeof(girl));
    while(scanf("%d%d",&x,&y)==2){
        if(x==-1&&y==-1)
            break;
        line[x][y]=1;
    }


    int sum=0;
    for(int i=m+1; i<=m+n; i++)
    {
        memset(used,0,sizeof(used));
        if(found(i)) sum++;
    }
    if(sum==0)
        printf("No Solution!\n");
    else{
        printf("%d\n",sum);
        for(int i=1;i<=m;i++){
            if(girl[i]!=0)
                printf("%d %d\n",i,girl[i]);
        }
    }
    return 0;
}
```
